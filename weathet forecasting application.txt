function fmtTime(unix,offset){
  const d=new Date((unix+offset)*1000);
  let h=d.getUTCHours(),m=d.getUTCMinutes();
  const ap=h>=12?'PM':'AM';
  h=h%12||12;
  return `${h}:${String(m).padStart(2,'0')} ${ap}`;
}

function dewPt(temp,hum){

  return ((b*al)/(a-al)).toFixed(1);
}

    ho:{en:'☀️ High heat — water crops early morning or evening',hi:'☀️ अधिक गर्मी — सुबह या शाम सिंचाई करें',te:'☀️ అధిక వేడి — పొద్దున లేదా సాయంత్రం నీళ్ళు పెట్టండి'},
    fr:{en:'🥶 Frost risk — protect sensitive crops',hi:'🥶 पाले का खतरा — नाजुक फसलें ढकें',te:'🥶 మంచు ప్రమాదం — సున్నితమైన పంటలు కప్పండి'},
    hw:{en:'💨 High winds — avoid pesticide/fertilizer spraying',hi:'💨 तेज हवा — कीटनाशक छिड़काव न करें',te:'💨 అధిక గాలి — పురుగుల మందు పిచికారీ చేయకండి'},
    hf:{en:'🍄 High humidity — monitor crops for fungal disease',hi:'🍄 उच्च आर्द्रता — फफूंद रोग की निगरानी करें',te:'🍄 అధిక తేమ — శిలీంధ్ర వ్యాధి గమనించండి'},
    gs:{en:'✅ Good conditions for pesticide/fertilizer application',hi:'✅ कीटनाशक/उर्वरक के लिए अच्छी स्थिति',te:'✅ పురుగుల మందు/ఎరువుల వాడుకకు మంచి పరిస్థితి'},
    th:{en:'⛈️ Thunderstorm — seek shelter, avoid open fields',hi:'⛈️ आंधी — खुले खेतों से दूर रहें',te:'⛈️ ఉరుముల వర్షం — ఆశ్రయం తీసుకోండి'}
  };
  if(['Rain','Drizzle'].includes(cond)){tips.push(T.ri[L]);tips.push(T.rh[L]);}
  if(cond==='Thunderstorm')tips.push(T.th[L]);
  if(temp>35)tips.push(T.ho[L]);
  if(temp<3)tips.push(T.fr[L]);
  if(wind>30)tips.push(T.hw[L]);
  if(hum>85&&!['Rain','Drizzle','Thunderstorm'].includes(cond))tips.push(T.hf[L]);
  if(!['Rain','Drizzle','Thunderstorm'].includes(cond)&&wind<20&&temp>10&&temp<30)tips.push(T.gs[L]);
  if(!tips.length)tips.push({en:'🌱 Normal conditions — routine farming is fine',hi:'🌱 सामान्य स्थिति — नियमित खेती ठीक है',te:'🌱 సాధారణ పరిస్థితులు — సాధారణ వ్యవసాయం సరే'}[L]);
  return tips;
}

function renderWeather(data){
  lastData=data;
  const{main,weather,wind,clouds,visibility,sys,name,timezone}=data;
  const cMain=weather[0].main;
  const cInfo=condMap[cMain]||{icon:'🌤️',bg:'wc'};
  const temp=Math.round(main.temp);
  const feels=Math.round(main.feels_like);
  const hum=main.humidity;
  const windKmh=Math.round(wind.speed*3.6);
  const vis=visibility?(visibility/1000).toFixed(1):'—';
  const dp=dewPt(main.temp,hum);
  const tz=timezone||0;
  const nowD=new Date((Date.now()/1000+tz)*1000);
  const timeStr=nowD.toUTCString().slice(17,22);
  const dateStr=nowD.toUTCString().slice(0,16);

  document.body.className=cInfo.bg;

  document.getElementById('cityName').textContent=name;
  document.getElementById('cityCountry').textContent=`${sys.country} · ${dateStr}`;
  document.getElementById('cityTime').textContent=`Local: ${timeStr}`;
  document.getElementById('weatherDesc').textContent=weather[0].description;
  document.getElementById('weatherIconBig').textContent=cInfo.icon;
  document.getElementById('tempVal').textContent=temp;
  document.getElementById('feelsLike').textContent=`${t('feels')} ${feels}°C`;
  document.getElementById('humVal').textContent=hum;
  document.getElementById('windVal').textContent=windKmh;
  document.getElementById('windUnitSpan').textContent=t('windUnit');
  document.getElementById('pressVal').textContent=main.pressure;
  document.getElementById('visVal').textContent=vis;
  document.getElementById('visBar').style.width=Math.min((parseFloat(vis)||0)/10*100,100)+'%';
  document.getElementById('cloudVal').textContent=clouds?.all??'—';
  document.getElementById('dewVal').textContent=dp;
  document.getElementById('sunriseTime').textContent=fmtTime(sys.sunrise,tz);
  document.getElementById('sunsetTime').textContent=fmtTime(sys.sunset,tz);

  const uvEst=cMain==='Clear'?(temp>30?'8-10 🔴':'5-7 🟡'):cMain==='Clouds'?'2-4 🟢':'1-2 🟢';
  document.getElementById('uviVal').textContent=uvEst;

  const clothes=getClothes(temp);
  document.getElementById('clothingBadges').innerHTML=clothes.map(c=>`<span class="rec-badge">${c}</span>`).join('');

  const umb=getUmbrella(cMain,hum);
  document.getElementById('umbrellaBadge').innerHTML=`<span class="rec-badge ${umb.type}" style="margin-top:10px">${umb.text}</span>`;
  document.getElementById('aiRecText').textContent=getAIText(temp,hum,cMain);

  const tips=getFarmerTips(temp,hum,windKmh,cMain);
  document.getElementById('farmerTips').innerHTML=tips.map(tip=>`<div class="farmer-tip"><span class="farmer-tip-text">${tip}</span></div>`).join('');

  document.getElementById('weatherMain').classList.add('active');
  document.querySelectorAll('[data-i18n]').forEach(el=>{
    const k=el.getAttribute('data-i18n');
    const v=strings[currentLang]?.[k];
    if(v)el.textContent=v;
  });
}

async function fetchWeather(){
  const city=document.getElementById('cityInput').value.trim();
  if(!city){showError('Please enter a city name.');return;}
  document.getElementById('errorBox').classList.remove('active');
  document.getElementById('weatherMain').classList.remove('active');
  showSpinner(true);
  try{
    const r=await fetch(`https://api.openweathermap.org/data/2.5/weather?q=${encodeURIComponent(city)}&appid=${API_KEY}&units=metric`);
    if(r.status===404){showSpinner(false);showError(t('notfound'));return;}
    if(!r.ok){showSpinner(false);showError(`API Error: ${r.status}`);return;}
    const d=await r.json();
    showSpinner(false);
    renderWeather(d);
  }catch(e){
    showSpinner(false);
    showError(t('network'));
  }
}
